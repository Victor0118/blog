``` python
# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def removeNthFromEnd(self, head, n):
        """
        :type head: ListNode
        :type n: int
        :rtype: ListNode
        """
        
        c = head
        s = head
        count = 0
        
        
        while c != None:
            c = c.next
            count += 1
            if count > n + 1:
                s = s.next
        
        if count == n:
            return head.next
        
        s.next = s.next.next
        return head
        
        

```
